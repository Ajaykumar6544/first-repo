import java.util.*;
class Solution{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int a[]=new int[n];
        for(int i=0;i<n;i++){
            a[i]=sc.nextInt();
        }
        
       Arrays.sort(a);
       int s=0;
       if(n%2==0){
        s=(a[n / 2 - 1] + a[n / 2]) / 2;
        System.out.print(s);
       }
       else{
        s=a[n/2];
        System.out.print(s);
       }
       
       
        }
    }
