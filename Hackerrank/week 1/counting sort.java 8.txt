import java.util.*;
class Solution{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int arr[]=new int[n];
        for(int i=0;i<n;i++){
            arr[i]=sc.nextInt();
        }
        int f[]=new int[100];
        for(int i=0;i<n;i++){
            f[arr[i]]++;
        }
        for(int i=0;i<100;i++){
            System.out.print(f[i]+" ");
        }
    }
}
